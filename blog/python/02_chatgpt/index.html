openAI提供のchatGPTを試してみた

By hkaji 12 January 2023 Update 12 January 2023

chatGPTを試してみた

最近巷で流行っているOpenAI1という研究所が提供しているチャットボットを試してみました。 chatGPT(Generative Pre-trained Transformer)は、学習済言語モデルが質問に回答してくれるサービスです。 学習済AIが回答するので、準備された回答集から選んでくれるのではなく、質問に合わせて幅広に回答してくれます。 日本語にも対応してます。

ArduinoでLチカを実行するには、次のようなスケッチを作成する必要があります。

int LED = 13;

void setup() {
pinMode(LED, OUTPUT);
}

void loop() {
digitalWrite(LED, HIGH);
delay(1000);
digitalWrite(LED, LOW);
delay(1000);
}

また、ArduinoとLEDの間に1KΩの抵抗を接続する必要があります。具体的な配線図は次のとおりです。

Arduino - 1KΩ - LED - GND

BME280から温度と湿度を取得するには、Arduinoで次のようなスケッチを書く必要があります。

#include "Adafruit_BME280.h"
Adafruit_BME280 bme;

void setup() {
Serial.begin(9600);
while (!Serial) {
;
}
if (!bme.begin()) {
Serial.println("Could not find a valid BME280 sensor, check wiring!");
while (1) {
;
}
}
}

void loop() {
Serial.print("Temperature = ");
Serial.print(bme.readTemperature());
Serial.println(" *C");
Serial.print("Humidity = ");
Serial.print(bme.readHumidity());
Serial.println(" %");
delay(2000);
}

import openai

# APIキーを取得
openai.api_key = "API_KEY"

#モデルを指定
model_engine = "text-davinci-002"

#指示（Prompt）を設定
prompt = "日本の首都はどこですか?"

#推論を実行
completions = openai.Completion.create(
    engine=model_engine,
    prompt=prompt,
    max_tokens=1024,
    n=1,
    stop=None,
    temperature=0.5,
)

#推論結果を出力
message = completions.choices[0].text
print(message)

package main

import (
	"context"
	"fmt"
	gogpt "github.com/sashabaranov/go-gpt3"
)

func main() {
	c := gogpt.NewClient("Your token")
	ctx := context.Background()

	req := gogpt.CompletionRequest{
//		Model: "ada",
		Model: "text-davinci-002",
		MaxTokens: 100,
		Prompt:    "日本の首都はどこですか?",
	}
	resp, err := c.CreateCompletion(ctx, req)
	if err != nil {
		return
	}
	fmt.Println(resp.Choices[0].Text)
}
